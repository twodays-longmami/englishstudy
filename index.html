  q.opts.forEach(o=>{

    const b=document.createElement("button");

    b.className="quiz-option";b.textContent=o.kr;

    b.onclick=()=>pick("q",o.id,b,q);

    el.appendChild(b);

  });

  document.getElementById("q-fb").textContent="";

  st.ans=false;

}


function startSentence(){

  st.mode="sentence";

  const pool=sfl(SENTENCES);

  st.qs=pool.map(item=>{

    const wrong=sfl(SENTENCES.filter(s=>s.id!==item.id)).slice(0,3);

    return{item,opts:sfl([...wrong,item]),corr:item.id};

  });

  st.qi=0;st.score=0;st.ans=false;

  show("sentence-screen");renderS();

}

function renderS(){

  const t=st.qs.length,i=st.qi;

  if(i>=t){showResult(st.score,t);return}

  const q=st.qs[i];

  document.getElementById("s-pf").style.width=`${(i/t)*100}%`;

  document.getElementById("s-ctr").textContent=`${i+1} / ${t}`;

  document.getElementById("s-score").textContent=`${st.score}점`;

  document.getElementById("s-kr").textContent=q.item.kr;

  const el=document.getElementById("s-opts");

  el.innerHTML="";

  q.opts.forEach(o=>{

    const b=document.createElement("button");

    b.className="quiz-option";b.textContent=o.en;

    b.onclick=()=>pick("s",o.id,b,q);

    el.appendChild(b);

  });

  document.getElementById("s-fb").textContent="";

  st.ans=false;

}


function pick(pfx,selId,btn,q){

  if(st.ans)return;

  st.ans=true;

  const isQ=pfx==="q";

  const allBtns=document.querySelectorAll(`#${pfx}-opts .quiz-option`);

  allBtns.forEach(b=>{

    b.disabled=true;

    const txt=b.textContent;

    const match=q.opts.find(o=>(isQ?o.kr:o.en)===txt);

    if(match&&match.id===q.corr)b.classList.add("correct");

    else if(b===btn&&match&&match.id!==q.corr)b.classList.add("wrong");

  });

  const ok=selId===q.corr;

  if(ok)st.score++;

  const fb=document.getElementById(`${pfx}-fb`);

  fb.textContent=ok?"정답 ✓":"오답 ✗";

  fb.style.color=ok?"#1B7A4A":"#C0392B";

  setTimeout(()=>{st.qi++;isQ?renderQ():renderS()},1100);

}


function showResult(s,t){

  show("result-screen");

  const p=t>0?s/t:0;

  const stars=p>=.8?"★★★":p>=.6?"★★☆":p>=.4?"★☆☆":"☆☆☆";

  const msg=p>=.8?"훌륭해요! 완벽에 가깝네요 🎉":p>=.6?"잘했어요! 조금만 더 연습해봐요 💪":p>=.4?"괜찮아요, 다시 도전해봐요!":"포기하지 말고 다시 해봐요 📖";

  document.getElementById("r-stars").textContent=st.mode==="flashcard"?"":stars;

  document.getElementById("r-score").textContent=`${s} / ${t}`;

  document.getElementById("r-sub").textContent=st.mode==="flashcard"?"알고 있는 단어":"정답";

  document.getElementById("r-msg").textContent=msg;

  document.getElementById("r-replay").onclick=()=>{

    if(st.mode==="flashcard")startFlashcard();

    else if(st.mode==="quiz")startQuiz();

    else startSentence();

  };

}
